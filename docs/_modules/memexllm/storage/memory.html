[docs]    def delete_thread(self, thread_id: str) -> bool:
        """Delete thread by ID"""
        if thread_id in self.threads:
            del self.threads[thread_id]
            return True
        return False



[docs]    def search_threads(self, query: Dict[str, Any]) -> List[Thread]:
        """
        Basic search implementation

        Only supports exact matching on metadata fields
        """
        results = []

        for thread in self.threads.values():
            match = True

            # Check metadata matches
            for key, value in query.get("metadata", {}).items():
                if key not in thread.metadata or thread.metadata[key] != value:
                    match = False
                    break

            if match:
                results.append(thread)

        return results
